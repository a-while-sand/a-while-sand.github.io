安装

全局安装

npm install -g pm2
# 或者
yarn global add pm2
启动

参数说明：

  • --watch：监听应用目录的变化，一旦发生变化，自动重启。如果要精确监听、不见听的目录，最好通过配置文件。
  • -i --instances：启用多少个实例，可用于负载均衡。如果-i 0或者-i max，则根据当前机器核数确定实例数目。
  • --ignore-watch：排除监听的目录/文件，可以是特定的文件名，也可以是正则。比如--ignore-watch="test node_modules "some scripts""
  • -n --name：应用的名称。查看应用信息的时候可以用到。
  • -o --output <path>：标准输出日志文件的路径。
  • -e --error <path>：错误输出日志文件的路径。
pm2 start app.js --watch -i 2

# 或

pm2 start ecosystem.config.js
查看进程列表
pm2 list

查看进程信息
# pm2 describe id [id 通过pm2 list]
pm2 describe 1

重启
pm2 restart ecosystem.config.js

停止

可以先通过pm2 list获取应用的名字（–name指定的）或者进程id。

# 停止特定的应用

pm2 stop app_name | app_id

# 停止所有应用

pm2 stop all
日志查看
pm2 logs

与nacos相关
  1. 以下以公司环境为例
  2. 先增加相关依赖，安装依赖时，需要增加以下配置
"dependencies": {
  "nacos": "^2.5.1",
  "pm2": "^5.3.0"
}
  1. 再编写启动类ecosystem.config.js，内容如下
module.exports = {
  apps: [{
    name: "cnbksy",
    watch: true,
    script: './api.js',
    autorestart: true,
    exec_mode: "fork",
    env: {
      "PORT": 4000,
      "NODE_ENV": "development"
    }
  }]
}

指定端口4000，并且启动文件指向api.js，常用修改项为

  • ip：为本机ip；
  • port：为本机端口
  • serverList：nacos地址，若nacos为集群则写集群地址
  • serviceName：前端在nacos中的名称
  • namespace：前端放nacos哪个命名空间
const pm2 = require('pm2')
const { NacosNamingClient } = require('nacos');

const ip = '172.16.20.248';
const port = 4000;

const serverList = '172.16.20.248:8848';
//const serverList = '172.16.20.34:26001,172.16.20.154:26001,172.16.20.154:36001';
const serviceName = 'front';
const namespace = 'bksy';

const username = 'nacos';
const password = 'nacos';

const logger = console;

// 创建 Nacos 客户端，并指定命名空间
const namingClient = new NacosNamingClient({
  logger,
  serverList: serverList,
  namespace: namespace,
  username: username,
  password: password
});

// 注册服务到 Nacos
async function registerServiceWithNacos() {
  try {
    await namingClient.ready();
    await namingClient.registerInstance(serviceName, { ip: ip, port: port });
    console.log('注册nacos成功', 'Service registered with Nacos successfully.');
  } catch (err) {
    console.error('注册nacos失败', 'Error registering service with Nacos:', err);
  }
}

pm2.connect(function (err) {
  console.log('开始连接pm2并且打印一下端口', process.env.PORT, process.env.NODE_ENV)
  if (err) {
    console.log("我连接出错了", err)
    process.exit(2)
  }
  pm2.start({
    script: './.output/server/index.mjs',
    name: 'bksy',
    port: '4000',
    instances: 10,
    exec_mode: "cluster",
    watch: true,
    autorestart: true
  }, function (err, apps) {
    if (err) {
      console.log("我创建进程出错了", err)
      return pm2.disconnect()
    }
  })
  // 注册nacos
  registerServiceWithNacos();
  // 定时检测重启
  setInterval(() => {
    pm2.list((err, list) => {
      if (list.length === 1 && list[0].name === 'cnbksy') {
        process.exit(0)
      }
    })
  }, 5000)
})
