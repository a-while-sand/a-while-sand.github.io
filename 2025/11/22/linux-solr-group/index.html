Solr中先分组再统计关键词的词频

假设需要的student的core中，按年级分组，找到学生名中含有坚的学生的个数。

第1步：student的结构假设如下

student{
  grade: "一年级",
  name: "张志坚"
}，{
  grade: "二年级",
  name: "李坚"
}
1、先按grade分组，然后查询按grade分组统计包含坚的数量
{
    "facet": { 
   		 	 "type": "terms",
   			 "field": "grade"
   		 }
 }
2、直接按grade分组，且统计包含坚的数量
{
    "facet": {
        "categories": {
            "type": "terms",
            "field": "grade",
            "facet": {"keyword_frequency": "sum(termfreq(name,'坚'))"
            }}}
 }
3、java代码实现
public class TermFrequencyUtil {

    public static final String termField = "grade";
    public static final String statField = "name";

    private TermFrequencyUtil(){};

    public static String buildFrequencyFacet(String termField, String statField, String statValue){
        String jsonFacet = "{"
                + "term_stats: {"
                + "  type: terms,"
                + "  field: %s,"
                + "  limit: 100,"
                + "  mincount: 1,"
                + "  facet: {"
                + "    term_count: \"sum(termfreq(%s,'%s'))\""
                + "  }"
                + "}"
                + "}";
        return String.format(jsonFacet, termField, statField, statValue);
    }

    /**
   
     * @param statValue
     * @return
     */
    public static String buildYearFrequencyFacet(String statValue){
        return buildFrequencyFacet(termField, statField, statValue);
    }
}
/**
 *  
 * @return Map<年级, 出现次数>
 */
public Map<String, Long> getJigeFrequencyByGrade() {
    Map<String, Long> result = new TreeMap<>(); 
    try {
        SolrQuery query = new SolrQuery();
        query.setQuery("LiteratureCategory:8");
        //query.setQuery(solrQuery);
        query.setRows(0); // 不返回文档，只获取统计
        // 构建JSON Facet
        String jsonFacet = TermFrequencyUtil.buildYearFrequencyFacet(keyword);
        query.setParam("json.facet", jsonFacet);

        QueryResponse response = solrClient.query(coreName, query);
        // 解析分面结果
        SimpleOrderedMap  facetMap = (SimpleOrderedMap) response.getResponse().get("facets");
        if (facetMap != null && facetMap.get("term_stats")!=null) {
            SimpleOrderedMap gradeStats = (SimpleOrderedMap) facetMap.get("term_stats");
            List<SimpleOrderedMap> buckets = (List<SimpleOrderedMap>) gradeStats.get("buckets");

            for (SimpleOrderedMap bucket : buckets) {
                String grade = String.valueOf(bucket.get("val"));
                Long count = ((Number) bucket.get("term_count")).longValue();
                result.put(grade, count);
            }
        }
    } catch (Exception e) {
        System.err.println("查询Solr失败: " + e.getMessage());
        e.printStackTrace();
    }

    return result;
}

private static void testGradeFrequency(SolrGradeTermFrequency counter) {
    long startTime = System.currentTimeMillis();

    Map<String, Long> result = counter.getJigeFrequencyByGrade();

    long endTime = System.currentTimeMillis();

    if (result.isEmpty()) {
        System.out.println("未找到相关数据");
        return;
    }

    System.out.println("各年级'及格'出现次数统计:");
    System.out.println("+--------+------------+");
    System.out.println("|  年级  |  出现次数  |");
    System.out.println("+--------+------------+");

    long total = 0;
    for (Map.Entry<String, Long> entry : result.entrySet()) {
        System.out.printf("| %-6s | %-10d |\n", entry.getKey(), entry.getValue());
        total += entry.getValue();
    }

    System.out.println("+--------+------------+");
    System.out.printf("| 总计   | %-10d |\n", total);
    System.out.println("+--------+------------+");
    System.out.printf("查询耗时: %d ms\n", (endTime - startTime));
}
